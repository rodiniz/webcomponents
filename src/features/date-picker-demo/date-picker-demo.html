Usage Example

<ui-date-picker format="DD/MM/YYYY" value="2026-02-26" min="2026-01-01" max="2026-12-31" ></ui-date-picker> <script> const picker = document.querySelector('ui-date-picker'); // Listen to date changes picker.addEventListener('date-change', (e) => { console.log('ISO Value:', e.detail.value); console.log('Formatted Value:', e.detail.formattedValue); console.log('Format:', e.detail.format); }); // Programmatic API picker.setValue('2026-03-15'); console.log(picker.getISOValue()); // "2026-03-15" console.log(picker.getFormattedValue()); // "15/03/2026" picker.clear(); </script>
